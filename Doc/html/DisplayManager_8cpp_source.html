140 uint16_t DisplayManager::updateAll() const {
141  uint16_t ret = 0, idx = 0;
142  byte outDisable = (outputEnablePinState == HIGH ? LOW : HIGH);
143 
144  digitalWrite(outputEnablePin, outputEnablePinState);
145 
146  // Reverse iteration to account for shift register serial update order
147  std::deque<DisplayGroup>::const_reverse_iterator beg = _groups.rbegin();
148  std::deque<DisplayGroup>::const_reverse_iterator end = _groups.rend();
149 
150  for (idx = 0; beg != end; ++beg, ++idx) {
151  if ((*beg).update() != 0) {
152  ret = idx;
153  }
154  }
155 
156  digitalWrite(outputEnablePin, outDisable);
157 
158  return ret;
159 }
