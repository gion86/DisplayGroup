37 int DisplayGroup::update() const {
38  if (_nDisplay == 0) {
39  return -1;
40  }
41 
42  // Turn off al the displays
43  if (!_enabled) {
44  std::vector<Display>::const_iterator beg = _displays.begin();
45  std::vector<Display>::const_iterator end = _displays.end();
46 
47  for (; beg != end; ++beg) {
48  (*beg).turnOff();
49  }
50 
51  return 0;
52  }
53 
54  if (!_value) {
55  return -2;
56  }
57 
58  // Quotient
59  uint16_t quot = *(_value) / 10;
60  uint16_t tempV = quot;
61  // Division counter
62  byte divCount = 1;
63  boolean filled = false;
64 
65  // Update the first display
66  _displays[0].update(*(_value) % 10);
67 
68  // Look for zero filling in heading when *(_value) is < 10
69  if (quot == 0) {
70  for (byte i = 1; i < _nDisplay; ++i) {
71  _displays[i].update(0);
72  }
73  filled = true;
74  }
75 
76  // Count up division by 10 updates display with the remainder
77  while (quot != 0 && divCount < _nDisplay) {
78  quot = tempV / 10;
79 
80  _displays[divCount].update(tempV % 10);
81  divCount++;
82 
83  tempV = quot;
84  }
85 
86  // Return -3 if the number cannot be displayed with the number of displays in
87  // the group
88  if (divCount == _nDisplay && quot != 0) {
89  return -3;
90  } else if (divCount < _nDisplay && !filled) {
91  // Look for zero filling in heading when *(_value) is > 10
92  for (byte i = divCount; i < _nDisplay; ++i) {
93  _displays[i].update(0);
94  }
95  }
96 
97  return 0;
98 }
